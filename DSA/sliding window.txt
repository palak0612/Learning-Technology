What is Sliding Window?

Instead of recalculating things for every subarray, we slide a window over the array and reuse previous work.

int maxSumSubarray(vector<int>& arr, int k) {
    int sum = 0;

    // first window
    for (int i = 0; i < k; i++) {
        sum += arr[i];
    }

    int maxSum = sum;

    // slide window
    for (int i = k; i < arr.size(); i++) {
        sum += arr[i];
        sum -= arr[i - k];
        maxSum = max(maxSum, sum);
    }

    return maxSum;
}

int minSubArrayLen(int k, vector<int>& arr) {
    int l = 0, sum = 0;
    int ans = INT_MAX;

    for (int r = 0; r < arr.size(); r++) {
        sum += arr[r];

        while (sum >= k) {
            ans = min(ans, r - l + 1);
            sum -= arr[l];
            l++;
        }
    }

    return ans == INT_MAX ? 0 : ans;
}
